var canJump = function(nums) {
   let path = 0; //起点是0
   if (nums.length === 1) return true; //如果只有一个数字肯定能跳到
   for (let i = 0; i <= path; i++) {
       path = Math.max(path, i + nums[i]) //从该点出发，能跳到多远的距离

       if (path >= nums.length - 1) return true;
       //一旦有达到数组长度的就可以直接返回true了
   }

   //走完都还没达到 就表示不符合
};
